# i18n Support

The Markdown WYSIWYG Editor supports integration with external i18n systems like **react-i18next** or **next-intl**.

## Integration with i18n Libraries

### With react-i18next

```tsx
import { MarkdownEditor, I18N_KEYS } from 'markdown-wysiwyg-editor';
import { useTranslation } from 'react-i18next';

function App() {
  const { t } = useTranslation();
  const [content, setContent] = useState('');

  const texts = {
    table: {
      rowOperations: t(I18N_KEYS.table.rowOperations),
      addRowAbove: t(I18N_KEYS.table.addRowAbove),
      addRowBelow: t(I18N_KEYS.table.addRowBelow),
      deleteRow: t(I18N_KEYS.table.deleteRow),
      columnOperations: t(I18N_KEYS.table.columnOperations),
      addColumnLeft: t(I18N_KEYS.table.addColumnLeft),
      addColumnRight: t(I18N_KEYS.table.addColumnRight),
      deleteColumn: t(I18N_KEYS.table.deleteColumn),
      deleteTable: t(I18N_KEYS.table.deleteTable),
      cancel: t(I18N_KEYS.table.cancel),
    },
  };

  return <MarkdownEditor value={content} onChange={setContent} texts={texts} />;
}
```

### With next-intl

```tsx
import { MarkdownEditor, I18N_KEYS } from 'markdown-wysiwyg-editor';
import { useTranslations } from 'next-intl';

function App() {
  const t = useTranslations();
  const [content, setContent] = useState('');

  const texts = {
    table: {
      rowOperations: t(I18N_KEYS.table.rowOperations),
      addRowAbove: t(I18N_KEYS.table.addRowAbove),
      addRowBelow: t(I18N_KEYS.table.addRowBelow),
      deleteRow: t(I18N_KEYS.table.deleteRow),

      deleteColumn: t(I18N_KEYS.table.deleteColumn),

  return <MarkdownEditor value={content} onChange={setContent} texts={texts} />;

```

### Direct Text Override (Without i18n Library)

You can also directly pass translated texts:

```tsx
import { MarkdownEditor } from 'markdown-wysiwyg-editor';

function App() {
  const [content, setContent] = useState('');

  const texts = {
    table: {
      rowOperations: '行の操作',
      addRowAbove: '上に行を追加',
      addRowBelow: '下に行を追加',
      deleteRow: '行を削除',
      columnOperations: '列の操作',
      addColumnLeft: '左に列を追加',
      addColumnRight: '右に列を追加',
      deleteColumn: '列を削除',
      deleteTable: 'テーブル全体を削除',
      cancel: 'キャンセル',
    },
  };

  return <MarkdownEditor value={content} onChange={setContent} texts={texts} />;
}
```

## Translation Keys

The `I18N_KEYS` constant provides all translation keys:

```typescript
import { I18N_KEYS } from 'markdown-wysiwyg-editor';

// Table context menu keys
I18N_KEYS.table.rowOperations        // 'markdown_editor.table.row_operations'
I18N_KEYS.table.addRowAbove          // 'markdown_editor.table.add_row_above'
I18N_KEYS.table.addRowBelow          // 'markdown_editor.table.add_row_below'
I18N_KEYS.table.deleteRow            // 'markdown_editor.table.delete_row'
I18N_KEYS.table.columnOperations     // 'markdown_editor.table.column_operations'
I18N_KEYS.table.addColumnLeft        // 'markdown_editor.table.add_column_left'
I18N_KEYS.table.addColumnRight       // 'markdown_editor.table.add_column_right'
I18N_KEYS.table.deleteColumn         // 'markdown_editor.table.delete_column'
I18N_KEYS.table.deleteTable          // 'markdown_editor.table.delete_table'
I18N_KEYS.table.cancel               // 'markdown_editor.table.cancel'
```

## Helper Hook (Optional)

Create a custom hook for cleaner usage:

```tsx
import { I18N_KEYS } from 'markdown-wysiwyg-editor';
import { useTranslation } from 'react-i18next';

export function useMarkdownEditorTexts() {
  const { t } = useTranslation();

  return {
    table: {
      rowOperations: t(I18N_KEYS.table.rowOperations),
      addRowAbove: t(I18N_KEYS.table.addRowAbove),
      addRowBelow: t(I18N_KEYS.table.addRowBelow),
      deleteRow: t(I18N_KEYS.table.deleteRow),
      columnOperations: t(I18N_KEYS.table.columnOperations),
      addColumnLeft: t(I18N_KEYS.table.addColumnLeft),
      addColumnRight: t(I18N_KEYS.table.addColumnRight),
      deleteColumn: t(I18N_KEYS.table.deleteColumn),
      deleteTable: t(I18N_KEYS.table.deleteTable),
      cancel: t(I18N_KEYS.table.cancel),
    },
  };
}

// Usage
function App() {
  const texts = useMarkdownEditorTexts();
  return <MarkdownEditor texts={texts} />;
}
```

## TypeScript Support

```typescript
import { ITexts } from 'markdown-wysiwyg-editor';

// Your translation object will have full type safety
const myTexts: Partial<ITexts> = {
  table: {
    addRowAbove: 'My custom translation',
    // TypeScript will autocomplete and validate all keys
  },
};
```
